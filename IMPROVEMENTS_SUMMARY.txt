C# PROGRAMMATIC USAGE:

// Using presets
var settings = ObfuscationPresets.Heavy;
var settings = ObfuscationPresets.Light;
var settings = ObfuscationPresets.Balanced;

// With progress callback
var onProgress = new Action<ObfuscationProgress>(progress => {
    Console.WriteLine($"[{progress.CurrentStage}] {progress.Message}");
});

IB2.Obfuscate("temp", "input.lua", settings,
    out string error, onProgress, validate: true);

// With validation
var validator = new ObfuscationValidator();
if (validator.ValidateAll("out.lua", originalSize)) {
    Console.WriteLine("✓ Output is valid!");
}

// Using metrics
var metrics = new PipelineMetrics();
metrics.StartTotal();
// ... do work ...
metrics.StopTotal();
Console.WriteLine(metrics.GetReport());
